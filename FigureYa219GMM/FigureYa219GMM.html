小丫编辑校验

需求描述

这篇文章里用了高斯混合模型，从10个候选miRNA筛出了8个，并且获得了每个miRNA的系数，计算riskscore。

出自https://www.thno.org/v10p8771.pdf

Figure 4. The pattern of AUC and 1023 logistic regression models were based on Gaussian finite mixture models. (A) The pattern of the logistic regression model correlated with the AUC scores and was identified by a Gaussian mixture. There are eight clusters of 1023 combinations.

应用场景

通过排列组合构建所有可能的逻辑回归并计算每个模型下的AUC，通过高斯混合模型聚类确定最优模型簇，并根据最高AUC得到最终预测模型。

# 排列组合构建可能的逻辑回归模型
flist <- list()
for (i in 1:length(deg)) {
  message("--generate ",i," miRNA model...")
  tmp <- combn(deg, m = i) # 选取miRNA数目为i情况下的排列组合
  for (j in 1:ncol(tmp)) {
   f <- as.formula(paste0(event," ~ ",paste(tmp[,j],collapse = " + "))) # 构建当前模型
   flist <- append(flist, f) # 1023个模型，即2的10次方-1个模型
  }
}

# 运行逻辑回归模型并计算AUC
df <- cbind.data.frame(tcga.sinfo[basal.tcga, event, drop = F],
                       scale(t(tcga.mirna[deg,basal.tcga])))
pred_LR <- list()
aucDF <- NULL
n.model <- 2^length(deg)-1
for (i in 1:n.model) {
  
  # 显示运行进程
  display.progress(index = i,
                   totalN = n.model)
  
  f <- flist[[i]] # 提出当前逻辑回归公式
  g <- setdiff(as.character(f),c(event,"~")) # 拆分公式得到当前公式里的miRNA
  g <- unlist(strsplit(g," + ",fixed = T))
  
  # 运行逻辑回归
  l <- glm(formula = f, 
           data = df, 
           family = "binomial", 
           control = list(maxit = 50), 
           na.action = na.exclude)
  
  # 预测TCGA乳腺癌的复发
  pred_LR[[i]] <- data.frame(prob = predict(l, newdata = df,type="response"),
                             group = df[,event],
                             stringsAsFactors = F)
  
  # 计算AUC
  a <- quiet(auc(pred_LR[[i]]$group,
             pred_LR[[i]]$prob))
  
  # 生成结果
  aucDF <- rbind.data.frame(aucDF,
                            data.frame(model = i, # 当前模型数目
                                       auc = a, # 当前模型的AUC
                                       n.mirna = length(g), # 当前模型包含的miRNA数目
                                       stringsAsFactors = F),
                            stringsAsFactors = F)
}
